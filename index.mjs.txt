// index.mjs
import 'dotenv/config'   // loads .env(.local) when using `node -r dotenv/config index.mjs`
import { streamText } from 'ai'

const apiKey = process.env.OPENAI_API_KEY || process.env.VERCEL_OIDC_TOKEN
if (!apiKey) {
  console.error('Missing credentials. Set OPENAI_API_KEY or VERCEL_OIDC_TOKEN in .env.local')
  process.exit(1)
}

const model = 'openai/gpt-5.5'
const prompt = 'Explain quantum computing in simple terms.'

try {
  // Many wrappers read OPENAI_API_KEY from process.env automatically.
  // If the `ai` function accepts an apiKey option, pass it; otherwise ensure OPENAI_API_KEY is set.
  const result = streamText({
    model,
    prompt,
    // apiKey // optionally pass apiKey if the function supports explicit API key
  })

  for await (const chunk of result.textStream) {
    process.stdout.write(chunk)
  }
  process.stdout.write('\n')
} catch (err) {
  console.error('Stream error:', err)
  process.exit(1)
}
